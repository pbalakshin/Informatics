% Перечисление строчными буквами русского алфавита (ГОСТ 2.105-95, 4.1.7)
\makeatletter
\AddEnumerateCounter{\asbuk}{\russian@alph}{щ}      % Управляем списками/перечислениями через пакет enumitem, а он 'не знает' про asbuk, потому 'учим' его
\makeatother

% Уровень 1: а), б), в)...
\setlist[enumerate,1]{%
	label=\asbuk{enumi}),% chktex 9
	leftmargin=*%
}% chktex 10

% Уровень 2: а), б), в)... (или при необходимости 1), 2)...)
\setlist[enumerate,2]{%
	label=\asbuk{enumii}),% chktex 9
	leftmargin=*%
}% chktex 10

% Уровень 3: 1), 2), 3)...
\setlist[enumerate,3]{%
	label=\arabic{enumiii}),% chktex 9
	leftmargin=*%
}% chktex 10

% Уровень 4 (маркированный): -- (тире)
% Используем короткое тире (endash) для ненумерованных списков (ГОСТ 2.105-95, пункт 4.1.7, требует дефиса, но так лучше смотрится)
\setlist[itemize,1]{%
	label={--},%
	leftmargin=*%
}%

% --- 3. Глобальные настройки списков ---
\setlist[enumerate]{nosep}
\setlist[itemize]{nosep}

% --- 3. Настройка description (ГОСТ 2.105-95) ---
% Избегает ошибки "Negative labelwidth"
\setlist[description]{%
	style=standard,%             % Стандартный стиль (метка в тексте)
	% labelwidth=3cm,%             % Явная ширина метки (подберите под ваши термины)
	% leftmargin=3.5cm,%           % Отступ слева (labelwidth + отступ после метки)
	labelsep=0.5cm,%             % Расстояние между меткой и текстом
	font=\normalfont\bfseries,%  % Метки жирным шрифтом (ГОСТ требует выделения терминов)
	noitemsep,%                  % Без дополнительных интервалов между пунктами
	topsep=0pt%                  % Без отступа перед списком
}%
